# 1️⃣ Baixar o pacote tar.gz

# 2️⃣ Extrair para /opt

# 3️⃣ Renomear (opcional, deixa mais organizado)
mv /opt/Postman /opt/postman

# 4️⃣ Criar link simbólico para facilitar execução
ln -s /opt/postman/Postman /usr/local/bin/postman

# 5️⃣ Remover o arquivo compactado
rm postman.tar.gz

VSCODE 
sudo apt update && sudo apt upgrade -y

# -----------------------------
# 1️⃣ Instalar Python
# -----------------------------
sudo apt install -y python3 python3-pip python3-venv

# -----------------------------
# 2️⃣ Instalar Node.js e TypeScript
# -----------------------------
curl -fsSL https://deb.nodesource.com/setup_lts.x | sudo -E bash -
sudo apt install -y nodejs
sudo npm install -g typescript

# -----------------------------
# 3️⃣ Instalar compiladores C/C++
# -----------------------------
sudo apt install -y build-essential gdb

# -----------------------------
# 4️⃣ Instalar dependências e VS Code
# -----------------------------
sudo apt install -y wget gpg

# Adicionar repositório Microsoft para VS Code
wget -qO- https://packages.microsoft.com/keys/microsoft.asc | gpg --dearmor > packages.microsoft.gpg
sudo install -o root -g root -m 644 packages.microsoft.gpg /usr/share/keyrings/
sudo sh -c 'echo "deb [arch=amd64 signed-by=/usr/share/keyrings/packages.microsoft.gpg] https://packages.microsoft.com/repos/code stable main" > /etc/apt/sources.list.d/vscode.list'

# Instalar VS Code
sudo apt update
sudo apt install -y code

# -----------------------------
# 5️⃣ Instalar extensões essenciais do VS Code
# -----------------------------
# Python e Jupyter
code --install-extension ms-python.python
code --install-extension ms-toolsai.jupyter

# C/C++
code --install-extension ms-vscode.cpptools

# Node.js/TypeScript
code --install-extension esbenp.prettier-vscode
code --install-extension dbaeumer.vscode-eslint
code --install-extension ms-vscode.vscode-typescript-next

# Extras úteis
code --install-extension ms-python.vscode-pylance      # IntelliSense avançado Python
code --install-extension eamodio.gitlens               # Git avançado
code --install-extension formulahendry.code-runner     # Rodar snippets de código rapidamente
code --install-extension vscode-icons-team.vscode-icons # Ícones no Explorer

# -----------------------------
# 6️⃣ Finalização
# -----------------------------
echo "=========================================="
echo "Instalação concluída!"
echo "VS Code, Python, Node.js, TypeScript, C/C++ e extensões essenciais instaladas."
echo "------------------------------------------"
echo "Lembre-se de verificar as versões:"
echo "Python: python3 --version"
echo "Node.js: node -v | npm -v | tsc -v"
echo "C/C++: gcc --version | g++ --version"
echo "VS Code: code --version"
echo "=========================================="

DOCKER DESKTOP 
# Add Docker's official GPG key:
sudo apt update
sudo apt install ca-certificates curl
sudo install -m 0755 -d /etc/apt/keyrings
sudo curl -fsSL https://download.docker.com/linux/ubuntu/gpg -o /etc/apt/keyrings/docker.asc
sudo chmod a+r /etc/apt/keyrings/docker.asc

# Add the repository to Apt sources:
sudo tee /etc/apt/sources.list.d/docker.sources <<EOF
Types: deb
URIs: https://download.docker.com/linux/ubuntu
Suites: $(. /etc/os-release && echo "${UBUNTU_CODENAME:-$VERSION_CODENAME}")
Components: stable
Signed-By: /etc/apt/keyrings/docker.asc
EOF

sudo apt update
mkdir -p /opt/installers
cd /opt/installers

sudo wget https://desktop.docker.com/linux/main/amd64/docker-desktop-amd64.deb
sudo apt install -y ./docker-desktop-amd64.deb
